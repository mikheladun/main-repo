1   package com.teksystems.eag;

3   import java.util.ArrayList;
4   import java.util.List;
5   
6   /**
7    * Class for shopping basket which contains items
8    * 
9    * @author Mikhel Adun
10   */
11  public class Basket {
12  
13      // list of items in basket
14      private List<Item> items = null;
15  
16      /**
17       * Default Constructor
18       */
19      public Basket() {
20      }
21  
22      /**
23       * Getter for items
24       * 
25       * @return
26       */
27      public List<Item> getItems() {
28          return (this.items == null ? new ArrayList<Item>() : this.items);
29      }
30  
31      /**
32       * Setter for items
33       * 
34       * @param item
35       */
36      public void addItem(Item item) {
37          // lazy initialize List of items
38          if (this.items == null) {
39              this.items = new ArrayList<Item>();
40          }
41          this.items.add(item);
42      }
43  
44      /**
45       * Returns formatted string of items in Basket
46       * 
47       * @param basket
48       */
49      @Override
50      public String toString() {
51          StringBuffer output = new StringBuffer();
52          for (Item item : this.getItems()) {
53              output.append(item);
54              output.append("\r\n");
55          }
56  
57          return (output.toString());
58      }
59  
60      /**
61       * Returns true if basket contains items otherwise false
62       * 
63       * @return true/false
64       */
65      public boolean isEmpty() {
66          return (items == null || items.isEmpty());
67      }
68  }
