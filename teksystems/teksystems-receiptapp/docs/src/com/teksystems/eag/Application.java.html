5   /**
6    * The Receipt application adds items to a shopping basket, calculates sales tax 
7    * including import duty tax for each item and prints out the quantity, 
8    * item name, price, sales taxes, and the total amount. 
9    * 
10   * Basic sales tax is applicable at a rate of 10% on all goods, except books,
11   * food, and medical products that are exempt. Import duty is additional sales
12   * tax applicable on all imported goods at a rate of 5%, with no exemptions.
13   * 
14   * When I purchase items I receive a receipt which lists the name of all items
15   * and their price (including tax), finishing with the total cost of the items,
16   * and the total amounts of sales taxes paid. The rounding rules for sales tax
17   * are that for a tax rate of n%, a shelf price of p contains (np/100 rounded up
18   * to the nearest 0.05) amount of sales tax.
19   * 
20   * @author Mikhel Adun
21   * 
22   */

25      // 10% basic sales tax on all goods except books, food, and medical
26      public static final double BASIC_SALES_TAX_RATE = 0.10;
27      // 5% import duty tax on all imported goods
28      public static final double IMPORT_DUTY_TAX_RATE = 0.05;
29      public static final DecimalFormat DECIMAL_FORMAT = new DecimalFormat("0.00");

73      /**
74       * The rounding rules for sales tax are that for a tax rate of n%, a shelf price of p contains
75       * (np/100 rounded up to the nearest 0.05) amount of sales tax.
76       * 
77       * @param tax
78       * @return
79       */
80      public static double roundTax(double tax)
81      {
82          //round off to nearest 0.05
83          tax = Math.ceil(tax * 20) / 20;
84  
85          return (tax);
86      }
87  
88      /**
89       * Calculates total sales taxes and total amount of items
90       */
91      public static Receipt createReceipt(Basket basket) {
92          Receipt receipt = new Receipt(basket);
93          receipt.calculateTaxesAndTotal();
94  
95          return (receipt);
96      }
97  
98      /**
99       * Calculates basic sales tax and import duty for item
100      */
101     public static double calculateTax(Item item) {
102         double tax = 0.0;
103 
104         // calculate import duty
105         double importDuty = 0.0;
106         if (item.isImported()) {
107             importDuty = item.getPrice() * IMPORT_DUTY_TAX_RATE;
108         }
109 
110         // calculate sales tax
111         double salesTax = 0.0;
112         if (!item.isExempt()) {
113             salesTax = item.getPrice() * BASIC_SALES_TAX_RATE;
114         }
115 
116         tax = roundTax(salesTax + importDuty);
117 
118         return (tax);
119     }
