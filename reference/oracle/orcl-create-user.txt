define user=&&user
define password=system
set verify off;

drop user &&user cascade;
drop tablespace &&user including contents and datafiles cascade constraints;

CREATE TABLESPACE &&user DATAFILE '&&user' SIZE 50M AUTOEXTEND ON EXTENT MANAGEMENT LOCAL AUTOALLOCATE SEGMENT SPACE MANAGEMENT AUTO;

CREATE USER &&user IDENTIFIED BY &&password DEFAULT TABLESPACE &&user TEMPORARY TABLESPACE TEMP;

GRANT CONNECT, RESOURCE, ALTER TABLESPACE TO &&user;
GRANT UNLIMITED TABLESPACE to &&user;
GRANT CREATE SESSION TO &&user;
GRANT CREATE DATABASE LINK TO &&user;
GRANT CREATE TYPE TO &&user;
GRANT CREATE ANY TABLE TO &&user;
GRANT CREATE ANY VIEW TO &&user;
GRANT CREATE PROCEDURE TO &&user;
GRANT CREATE SEQUENCE to &&user;
GRANT CREATE ANY SEQUENCE to &&user;
GRANT CREATE TRIGGER to &&user;
GRANT CREATE MATERIALIZED VIEW to &&user;
GRANT SELECT ON SYS.V_$PARAMETER TO &&user;
GRANT QUERY REWRITE to &&user;

GRANT EXECUTE ON CTX_DDL TO &&user;
GRANT EXECUTE ON dbms_crypto TO &&user;
GRANT EXECUTE ON utl_raw TO &&user;
GRANT EXECUTE ON dbms_ddl TO &&user;

GRANT EXP_FULL_DATABASE TO &&user;
GRANT IMP_FULL_DATABASE TO &&user;

GRANT READ,WRITE ON DIRECTORY SYS.EXPORT_DIR TO &&user;

GRANT DBA TO &&user;

exit;
